# Translation file for the sMRI modality of recording bcgs

# Description: Translates sMRI filenames of recording `bcgs`
# Documentation: name_translation_files.txt 


[technique_id]
# Translation from file name to technique id
regexp="s%^.+(T1|B0|B1|SmartBrain).+$%$1%"

[technique_id_map]
T1=t1
B0=b0
B1=b1
SmartBrain=sbrain

[file_id]
# This regular expression matches all files to which links might be generated
# We believe that only the .nii files will be used so we link only to those
regexp =".+/smri/.+\.nii$"
