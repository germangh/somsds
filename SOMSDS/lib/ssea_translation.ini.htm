# Translation rules
[subject_id]
regexp="s|.*sleepstudy_(\d\d).*$|00$1|"


[modality_id]
regexp="s%^.+/(structural|functional|behavior)/.+\..+$%$1%"

[modality_id_map]
functional=fmri
structural=smri

[technique_id]
regexp="s%^.+/(structural|functional)/.+$+%$1%"

[technique_id_map]
structural=t1
functional=epi

[condition_id]
regexp="s%.*(switch|memory_retr|memory_enc|tol|fluency|encoding|retrieval).*%$1%"

[condition_id_map]
memory_enc=encoding
memory_retr=retrieval

[meta_id]
regexp="s%^.+(sleepstudy_\d\d|structural)_(bl|fu)_(switch|memory|tol|fluency|encoding|retrieval|)_*([^\.]*)\..+$%$4%"

[session_id]
regexp="s%.+_(bl|fu).+%$1%"

[file_id]
# Regular expression that matches the file names of files to which a symbolic 
# link needs to be generated
regexp ="^.+/(fluency|memory|switch|tol|functional|originals)/[^\._][^/]+(\.txt|\.nii|\.nii\.gz|\.nii)$"
