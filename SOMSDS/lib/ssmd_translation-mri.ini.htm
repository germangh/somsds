# Description: Import mri data into recording ssmd
# Documentation: name_translation_files.txt


# Labeling rules
[subject_id]
regexp="s|^.+/ssm_(\d\d\d\d)/.+\..+$|$1|"

[modality_id]
regexp="s%^.+(fmri|smri).*\..+$%$1%"

[technique_id]
regexp="s%^.+(epi|T1|dti_B0|dti|B0_phas|B0_magn|BO_phas|BO_magn).+$%$1%"

[technique_id_map]
T1=t1
B0=b0
B0_phas=b0-phase
B0_magn=b0-magn
dti_B0=dti-b0
BO_phas=b0-phase
BO_magn=b0-magn

[condition_id]
regexp="s%^.+/[^/]+epi_(auod_rest|taod_rest|rest|auod|taod)\..+$%$1%"

[condition_trans]
auod_rest=auod-rest
taod_rest=taod-rest

[meta_id]
regexp="s%^.+_smri_T1_([^/.]+)(\.nii.gz)$%$1%"

# Regular expression that matches the file names of files to which a symbolic 
# link needs to be generated
[file_id]
regexp ="^.+/ssm_\d\d\d\d/preproc/mri/.+(\.nii.gz|\.nii|\.bval|\.bvec)$"
