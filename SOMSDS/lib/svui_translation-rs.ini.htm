# Translation rules
[subject_id]
regexp="s|.*Sleep..\.S.{0,1}(\d\d\d)\..+$|0$1|"

[modality_id]
regexp="s%^.+/(eeg)/.+\..+$%$1%"

[device_id]
# All files were obtained with the egi256 device
regexp="s%.+%egi256%"

[condition_id]
regexp="s%^.+/(RS)/(EC|EO)/.+(SD|S).\.raw$%$1-$2-$3%"

[condition_id_map]
# Mapping from old condition IDs to new condition ids
RS-EC-SD=rs-ec-sd
RS-EC-S=rs-ec-ns
RS-EO-SD=rs-eo-sd
RS-EO-S=rs-eo-ns

[session_id]
regexp="s%.+(.)\.raw$%$1%"

[session_id_map]
N=n

# Regular expression that matches the file names of files to which 
# a symbolic link needs to be generated
[file_id]
regexp ="^.*import/.+/eeg/RS/.+\.raw$"
