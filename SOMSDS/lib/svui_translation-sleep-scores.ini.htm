# Translation rules
[subject_id]
regexp="s|.+S\d{0,1}(\d\d\d)[^/]*\..+$|0$1|"

[modality_id]
regexp="s%.+$%eeg%"

[device_id]
# All files were obtained with the egi256 device
regexp="s%.+%egi256%"

[condition_id]
regexp="s%.+%sleep%"

[session_id]
regexp="s%.+%scores%"

[block_id]
regexp="s%^.+_\d{0,1}(\d)(\.mat|\.dat)$%$1%"

[meta_id]
regexp="s%.+%fasttb%"

# Regular expression that matches the file names of files to which 
# a symbolic link needs to be generated
[file_id]
regexp ="^.*import/sleep-scores.+(\.dat|\.mat)$"
