# Description: Import mri data into recording vigd
# Documentation: name_translation_files.txt


# Labeling rules
[subject_id]
regexp="s|^.+/vigi_(\d\d\d\d)/.+\..+$|$1|"

[modality_id]
regexp="s%^.+(fmri|smri).*\..+$%$1%"

[technique_id]
regexp="s%^.+(epi|T1).+$%$1%"

[technique_id_map]
T1=t1

[condition_id]
regexp="s%^.+/[^/]+fmri_(rw|sd)_epi_(ctt)\..+$%$1-$2%"

[meta_id]
regexp="s%^.+(brain|trunc)\.nii\.gz$%$1%"

# Regular expression that matches the file names of files to which a symbolic 
# link needs to be generated
[file_id]
regexp ="^.+/vigi_\d\d\d\d/preproc/.+(\.nii.gz|\.nii)$"
